class ExpertSystemExecutor {
    constructor(matrixJson) {
        this.matrix = matrixJson;
        this.embedder = null;
    }

    async init() {
        // Загружаем ту же легкую мультиязычную модель
        this.embedder = await pipeline('feature-extraction', this.matrix.meta.model_used);
    }

    async evaluateState(currentStateText, contextVariables = {}) {
        // 1. Векторизуем текущее состояние (входной сигнал на любом языке)
        const stateOutput = await this.embedder(currentStateText, { pooling: 'mean', normalize: true });
        const stateVector = stateOutput.data;

        let triggeredActions = [];

        // 2. Self-Attention / Скалярное произведение по всей матрице эмбеддингов
        for (const [condId, condVector] of Object.entries(this.matrix.embeddings_layer)) {
            const score = cosine_similarity(stateVector, condVector);
            
            // Находим логическое правило для этого условия
            const rule = this.matrix.logic_gate_matrix.find(r => r.condition_id === condId);
            
            if (rule && score >= rule.required_score) {
                triggeredActions.push({
                    actionId: rule.action_id,
                    priority: rule.priority,
                    score: score
                });
            }
        }

        // 3. Разрешение конфликтов: сортировка по приоритету и уверенности (score)
        triggeredActions.sort((a, b) => b.priority - a.priority || b.score - a.score);

        if (triggeredActions.length > 0) {
            return this.executeAction(triggeredActions[0].actionId, contextVariables);
        }
        
        return "No action triggered";
    }

    async executeAction(actionId, ctx) {
        const action = this.matrix.action_registry[actionId];
        if (!action) return;

        console.log(`Executing Action ID: ${actionId} (${action.type})`);

        if (action.type === 'js_code') {
            // Безопасное или изолированное исполнение кода
            // В браузере можно использовать запуск через кастомный скоуп или Web Worker
            const execute = new Function('ctx', `return (async () => { ${action.payload} })()`);
            return await execute(ctx);
        }
        
        return action.payload;
    }
}

    constructor() {
        this.conditions = []; // [{id, text, threshold}]
        this.actions = [];    // [{id, type, command}]
        this.relations = [];  // [{id, cond_id, act_id, priority}]
        this.embedder = null;
    }

    // Инициализация мультиязычной модели в браузере
    async initModel() {
        this.embedder = await pipeline('feature-extraction', 'Xenova/multilingual-e5-small');
    }

    // Добавление условия с проверкой на дубликаты смыслов
    async addCondition(text, threshold = 0.85) {
        const output = await this.embedder(text, { pooling: 'mean', normalize: true });
        const newEmbedding = output.data; // Float32Array

        // Проверяем коллизии смыслов с уже существующими
        for (let cond of this.conditions) {
            const similarity = cosineSimilarity(newEmbedding, cond.embedding);
            if (similarity > 0.92) {
                throw new Error(`Дублирование смысла! Слишком похоже на: "${cond.text}" (Сходство: ${similarity.toFixed(2)})`);
            }
        }

        const id = `cond_${Date.now()}`;
        this.conditions.push({ id, text, threshold, embedding: newEmbedding });
        return id;
    }

    addAction(type, command, desc = "") {
        const id = `act_${Date.now()}`;
        this.actions.push({ id, type, command, description: desc });
        return id;
    }

    bind(condId, actId, priority = 1) {
        this.relations.push({ cond_id: condId, action_id: actId, priority });
    }

    // Сборка финального бинарно-текстового пакета знаний
    compileMatrix() {
        const exportPack = {
            meta: { total_rules: this.conditions.length },
            // Упаковываем веса в один массив для быстрой загрузки в WebGL/WebGPU исполнителем
            weights: {}, 
            rules: [],
            commands: {}
        };

        this.conditions.forEach(c => {
            // Конвертируем Float32Array в обычный массив для JSON
            exportPack.weights[c.id] = Array.from(c.embedding); 
        });

        this.relations.forEach(r => {
            const cond = this.conditions.find(c => c.id === r.cond_id);
            exportPack.rules.push({
                c_id: r.cond_id,
                a_id: r.action_id,
                th: cond.threshold,
                pr: r.priority
            });
        });

        this.actions.forEach(a => {
            exportPack.commands[a.id] = { t: a.type, cmd: a.command };
        });

        return JSON.stringify(exportPack);
    }

        for (let stateText of currentStatesArray) {
            const out = await this.embedder(stateText, { pooling: 'mean', normalize: true });
            stateVectors.push(out.data);
        }

        const activeConditions = [];

            for (let stateVec of stateVectors) {

                if (sim > maxSimilarity) maxSimilarity = sim;
            }
            if (maxSimilarity > 0.80) { // Порог распознавания смысла
                activeConditions.push({ cond, score: maxSimilarity });
            }
        }

            for (let active of activeConditions) {

                totalActionScore += bridgeScore * active.score;

            this.executeCode(bestAction.code);
        }
    }

    executeCode(code) {
        const run = new Function(code);
        run();
    }

Прямо во время ввода, как только срабатывает регулярное выражение или стеммер по словарю акторов, слово оборачивается в цветной бейдж.Пример: Текст Робот ждет на складе превращается в [Робот (синий)] ждет на [складе (зеленый)].Колонка «Авто-Теги»: Список распознанных сущностей, который обновляется на лету (debounced ввод, чтобы не спамить процессор).Специфичные колонки:Для Условий: Порог активации (слайдер от 0.0 до 1.0, по умолчанию 0.85).Для Действий: Поле ввода кода (Code Editor с подсветкой синтаксиса для JS-команд).Детализация Правой Части: Дашборд, Контроль и ТестыЭто динамический «кокпит» компилятора. Он разделен на вертикальные блоки.Блок 1: Конфигурация и Сборка (Top Bar)Выбор модели: Выпадающий список моделей из Hugging Face, доступных в Transformers.js (например, Xenova/multilingual-e5-small, Xenova/paraphrase-multilingual-MiniLM-L12-v2).Кнопка «Скомпилировать Матрицу»: Запускает процесс токенизации и генерации эмбеддингов для всех строк из левой части. Рядом отображается статус-бар и размер финального JSON-файла (например, «Сборка успешна: 142 Кб»).Кнопка «Экспорт»: Скачивание готового файла знаний для Исполнителя.Блок 2: Семантический дашборд (Визуализация матрицы)До компиляции он пуст. Ячейки на пересечении показывают степень семантической близости между ними.Если эксперт случайно создал два слишком похожих условия, ячейка загорится ярко-красным (сходство \(>0.90\)). Нажатие на ячейку подсветит эти строки в левой таблице.Граф Смысловых Связей: Визуальное облако (кластеры), где точки — это условия и действия. Точки созвучных сущностей (например, всё, что связано с движением и складом) будут физически сгруппированы рядом.Блок 3: Модуль Песочницы и Тестирования (Sandbox Runtime)Позволяет протестировать «мозг» системы прямо внутри Билдера перед экспортом.Поле ввода «Текущее состояние мира»: Сюда эксперт может вписать любую сырую фразу от руки (даже на другом языке). Например: "Тележка приехала на зарядку, пустая".Кнопка «Запустить тест»: Имитирует один тик Исполнителя.Окно результатов (Победившее действие): Выводит крупным шрифтом действие, которое выбрал трансформер, и процент уверенности. Например: Выбрано действие: Робот запрашивает зарядку (Уверенность: 91%).Ранжированный список: Ниже выводятся все остальные действия и условия с их баллами внимания, чтобы эксперт видел, почему система приняла именно такое решение.Блок 4: Консоль (Bottom Logs)Черное окно системных логов в самом низу правой части. Преимущества построчного текстового подходаНевероятная скорость прототипирования: Эксперт может просто скопировать готовое текстовое описание ТЗ из документации, вставить его в левое окно, за пару минут расставить дефисы и маркеры -> для кода — и "мозг" робота готов.Контроль версий (Git-friendly): Вся база знаний — это теперь один плоский текстовый файл (.txt или .md). Его можно хранить в Git, смотреть историю изменений (diff), видеть, кто и какое условие изменил. В случае классических таблиц или баз данных это делать крайне неудобно.Плавный UX: Редактор слева налету подсвечивает синтаксис (прямо в тексте красит найденных актеров в их цвета), а правая часть мгновенно обновляет счетчики сущностей и выдает предупреждения, если в коде действия допущена синтаксическая ошибка.Если этот текстовый подход принимаем за основу Билдера, давайте решим: хотите ли вы увидеть структуру и код такого JS-парсера, который превратит этот текст в готовые объекты для Transformer.js, или мы перейдем к проектированию синтаксиса сложных условий (например, как в таком тексте изящно задавать индивидуальные пороги чувствительности для конкретного условия)?
